298     private void boutonSaveActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_boutonSaveActionPerformed
299         // TODO add your handling code here:
300     	
301         if(jsonActive) {
302         	CustomFileChooser jsonFileChooser = new CustomFileChooser("json", "Choisir un nom de fichier JSON pour la sauvegarde");
303         	jsonFileChooser.showSaveDialog(this);
304         	if(jsonFileChooser.getSelectedFile() != null) {
305         		xmlPath = jsonFileChooser.getSelectedFile().getAbsolutePath();
306         		Saver.save(xmlPath+".json", textXml);
307         	}
308         }
309         
310         else {
311         	CustomFileChooser xmlFileChooser = new CustomFileChooser("xml", "Choisir un nom de fichier XML pour la sauvegarde");
312         	xmlFileChooser.showSaveDialog(this);
313         	if(xmlFileChooser.getSelectedFile() != null) {	
314             	jsonPath = xmlFileChooser.getSelectedFile().getAbsolutePath();
315             	Saver.save(jsonPath+".xml", textJson);
316         	}
317         }
